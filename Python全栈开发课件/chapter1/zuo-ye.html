def sushu():
    # 定义返回结果
    result= []
    for i in range(100,200):
        # 默认都是素数
        t_or_f =False
        #从2作为除数开始判断当前数是否能够被除1和本身的其他数整除
        for j in range(2,int(i/2)):
            # 如果能被2或者小于自己的数整除，这不是素数，改变t_or_f 为True
            if i%j==0:
                t_or_f = True
                break
        if t_or_f ==False:
            result.append(i)
    return result

def bubbleSort(arr):
    n = len(arr)

    # 遍历所有数组元素
    for i in range(n):

        # Last i elements are already in place
        for j in range(0, n-i-1):

            if arr[j] > arr[j+1] :
                arr[j], arr[j+1] = arr[j+1], arr[j]

def index():
    n = 2
    count = 0
    while True:
        for j in range(2,n):
            if n%j == 0:
                n = n+1
                break
        else:
            # print('%s是素数'%(n))
            yield n
            n = n + 1
            count += 1
        if count >= 10:
            break


a = index()
for i in range(2):

import time

def box(f):   # 用于计算一个函数运行时间
    def waste():
        start = time.time()  # 函数开始时间
        for i in range(5):
            f()  # 调用时间
        end = time.time()  # 函数结束时间
        print((end - start)/5)
    return waste

    @classmethod
    def set_country(cls, country):  # 类方法set_country

    def get_country(cls):  # 类方法get_country

    def set_money(self, value):  # 私有属性的set
        self.__money = value

    def get_money(self):  # 私有属性的get
        return  self.__money

    @staticmethod
    def fun(a):  # 静态方法方法完成对一个列表进行冒泡排序
