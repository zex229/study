餐厅

menu = [('北京烤鸭', 198), ('扬州炒饭', 20), ('京酱肉丝', 30)]
select =[]
msg = []
while True:
    print("""
    ****欢迎到八维餐厅****
    1、菜品展示
    2、点餐
    3、结账
    4、留言
    -1、退出系统
    """)
    n = input('请输入选择：')
    if n == '1':
        print('菜品展示')
        for i in enumerate(menu):
            print(i)
    elif n == '2':
        print('点餐')
        for i in enumerate(menu):
            print(i)
        while True:
            n = int(input('请输入号码'))
            select.append(menu[n])
            b = input('是否继续选择-1退出')
            if b == '-1':
                break
        print('选择菜品', select)
    elif n == '3':
        print('结账')
        t = 0
        for i in select:
            t += i[1]
        print('需要支付： ', t)
        while True:
            m = int(input('请输入金额：'))
            if m < t:
                print('金额不足！')
                continue
            else:
                print('支付成功！')
                break
    elif n == '4':
        print('留言')
        m = input('请输入留言')
        msg.append(m)
        print(msg)
    elif n == '-1':
        print('退出系统')
        break
    else:
        print('输入错误！')
